    <script>
        function convertToGrams() {
            const cups = parseFloat(document.getElementById('cups').value);
            if (isNaN(cups) || cups <= 0) {
                alert('Please enter a valid number of cups.');
                return;
            }
            const grams = cups * 120; // Example: 1 cup of flour is approximately 120 grams
            document.getElementById('grams').value = grams;
        }

        function convertToCups() {
            const grams = parseFloat(document.getElementById('grams').value);
            if (isNaN(grams) || grams <= 0) {
                alert('Please enter a valid number of grams.');
                return;
            }
            const cups = grams / 120; // Example: 1 cup of flour is approximately 120 grams
            document.getElementById('cups').value = cups;
        }
    </script>
